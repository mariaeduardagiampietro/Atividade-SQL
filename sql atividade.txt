SELECT * FROM VETERINARIO 
WHERE ESPECIALIDADE = 'Dermatologista'


SELECT * FROM PACIENTE p
WHERE p.IDADE BETWEEN 2 AND 5


SELECT * FROM AGENDAMENTO  
WHERE DATA_HORA BETWEEN '2024-01-01' AND '2024-03-31'


SELECT * FROM MEDICAMENTOS m 
WHERE m.PRECO > 25


SELECT * FROM RECEITA r 
WHERE r.DATA_EMISSAO > '2024-06-01' 
AND r.ID_AGENDAMENTO IN (
	SELECT r.ID_AGENDAMENTO FROM AGENDAMENTO a 
	WHERE a.ID_VETERINARIO = (
		SELECT ID_VETERINARIO FROM VETERINARIO v
		WHERE v.NOME = 'Dr. Ana Souza'
		)
	)
	
	
SELECT * FROM PACIENTE p
WHERE ID_TUTOR = ( 
	SELECT id_tutor FROM TUTOR
	WHERE nome = 'Maria Silva')

                 
SELECT * FROM ITENSRECEITA
WHERE ID_MEDICAMENTO =  ( 
	SELECT id_medicamento FROM MEDICAMENTOS m 
	WHERE NOME = 'Anti- Inflamatório Y')
                 
	
SELECT * FROM AGENDAMENTO
WHERE ID_VETERINARIO = (
	SELECT id_veterinario FROM VETERINARIO 
	WHERE NOME = 'Dr. Pedro Lima') 
	AND MOTIVO = 'Exame de sangue'
	
	
SELECT * FROM MEDICAMENTOS
WHERE DESCRICAO IS NOT NULL AND PRECO < 50


SELECT * FROM PACIENTE
WHERE PESO BETWEEN 3 AND 10

                 
SELECT * FROM PACIENTE
WHERE ID_TUTOR = (
	SELECT ID_TUTOR FROM TUTOR 
	WHERE NOME = 'Carlos Lima') 
	AND PESO > 5

WHERE ESPECIE = 'Cão'
AND PESO < '7'

(SELECT ID_PACIENTE FROM AGENDAMENTO)
